    <h1>Chemical Test Application</h1>


    <h2>Available Tests:</h2>

    <ul>

        <li>flame</li>

        <li>sodium hydroxide</li>

        <li>silver nitrate solution</li>

        <li>dilute acid</li>

        <li>barium chloride solution</li>

    </ul>


    <h2>Enter Test Category:</h2>

    <input type="text" id="test_category">

    <button onclick="ask_question()">Ask Question</button>


    <h2>Responses:</h2>

    <ul id="responses"></ul>


    <h2>Guess the Ions:</h2>

    <input type="text" id="positive_ion" placeholder="Guess Positive Ion">

    <input type="text" id="negative_ion" placeholder="Guess Negative Ion">

    <button onclick="check_guesses()">Check Guesses</button>


    <script type="text/python">

        from browser import document, alert

        import random


        list1 = ['lithium', 'sodium', 'potassium', 'calcium', 'copper (II)', 'magnesium', 'aluminium', 'iron (II)', 'iron (III)']

        list2 = ['chloride', 'bromide', 'iodide', 'carbonate', 'sulfate']


        item1 = random.choice(list1)

        item2 = random.choice(list2)


        question_count = 0

        correct_count = 0


        def answer_question(item, category):

            flame = {

                'lithium': 'You observe a crimson flame',

                'sodium': 'You observe a yellow flame',

                'potassium': 'You observe a lilac flame',

                'calcium': 'You observe a brick-red flame',

                'copper (II)': 'You observe a blue-green flame',

            }


            sodium_hydroxide = {

                'magnesium': 'A white precipitate forms',

                'calcium': 'A white precipitate forms',

                'aluminium': 'A white precipitate forms, then dissolves in solution',

                'copper (II)': 'A blue precipitate forms',

                'iron (II)': 'A dirty green precipitate forms',

                'iron (III)': 'A reddish brown precipitate forms'

            }


            silver_nitrate_solution = {

                'chloride': 'A white precipitate forms',

                'bromide': 'A cream precipitate forms',

                'iodide': 'A yellow precipitate forms',

            }


            dilute_acid = {

                'carbonate': 'CO₂ is produced',

            }


            barium_chloride_solution = {

                'sulfate': 'A white precipitate forms',

            }


            if category == 'flame':

                return flame.get(item)

            elif category == 'sodium hydroxide':

                return sodium_hydroxide.get(item)

            elif category == 'silver nitrate solution':

                return silver_nitrate_solution.get(item)

            elif category == 'dilute acid':

                return dilute_acid.get(item)

            elif category == 'barium chloride solution':

                return barium_chloride_solution.get(item)

            else:

                return None


        def ask_question():

            global question_count

            category = document['test_category'].value.strip().lower()

            question_count += 1


            answer1 = answer_question(item1, category)

            answer2 = answer_question(item2, category)


            responses = []

            if answer1 or answer2:

                if answer1:

                    responses.append(answer1)

                if answer2:

                    responses.append(answer2)

            else:

                responses.append("You observe no changes.")


            document['responses'].clear()  # Clear previous responses

            for response in responses:

                document['responses'] <= f"<li>{response}</li>"


        def check_guesses():

            global correct_count

            guess1 = document['positive_ion'].value.strip().lower()

            guess2 = document['negative_ion'].value.strip().lower()


            response = ""

            if guess1 == item1:

                response += f"Correct! The positive ion is indeed {item1}.<br>"

                correct_count += 1

            else:

                response += f"Incorrect. The positive ion was {item1}.<br>"


            if guess2 == item2:

                response += f"Correct! The negative ion is indeed {item2}.<br>"

                correct_count += 1

            else:

                response += f"Incorrect. The negative ion was {item2}.<br>"


            score = correct_count * 10 - question_count

            score = max(score, 0)


            response += f"Your score is: {score}/20"

            alert(response)

    </script>
